<script>
    const apiKey = ""; 
    const taskContainer = document.getElementById('task-container');
    const addTaskBtn = document.getElementById('add-task-btn');

    addTaskBtn.addEventListener('click', async () => {
        const name = document.getElementById('task-name').value;
        const importance = document.getElementById('importance').value;
        const urgency = document.getElementById('urgency').value;
        const time = parseInt(document.getElementById('estimated-time').value) || 0;

        if (!name) return;

        const realTime = Math.ceil(time * 1.2);
        const taskId = Date.now();
        
        createTaskElement(taskId, name, importance, urgency, realTime);
        document.getElementById('task-name').value = '';
        
        await generateSubtasks(taskId, name);
    });

    function createTaskElement(id, name, importance, urgency, time) {
        const priorityLabel = (importance === 'high' && urgency === 'high') ? '🔥 Do First' :
                             (importance === 'high' && urgency === 'low') ? '📅 Schedule' :
                             (importance === 'low' && urgency === 'high') ? '🤝 Delegate' : '🗑️ Don\'t Do';

        const html = `
            <div id="task-${id}" class="notion-card p-4 border-l-4 ${importance === 'high' ? 'border-blue-500' : 'border-gray-300'}" data-task-name="${name}">
                <div class="flex justify-between items-start">
                    <div>
                        <h3 class="font-bold text-lg task-title">${name}</h3>
                        <div class="text-xs text-gray-500 mt-1">Status: ${priorityLabel} | Plan: ${time} Min</div>
                    </div>
                    <button onclick="copyToNotion(${id})" class="text-xs bg-gray-100 hover:bg-gray-200 px-2 py-1 rounded">📋 Copy for Notion</button>
                </div>
                <div id="subtasks-${id}" class="mt-4 pl-4 border-l border-gray-200 space-y-1">
                    <div class="loader-container text-xs text-gray-400 italic">
                        <div class="loader"></div> Denke nach...
                    </div>
                </div>
            </div>
        `;
        taskContainer.insertAdjacentHTML('afterbegin', html);
    }

    async function generateSubtasks(taskId, taskName) {
        const subtaskContainer = document.getElementById(`subtasks-${taskId}`);
        const systemPrompt = "Zerlege die Aufgabe in max. 4 Häppchen. Antworte NUR mit einer JSON-Liste von Strings.";

        try {
            const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    contents: [{ parts: [{ text: `Task: ${taskName}` }] }],
                    systemInstruction: { parts: [{ text: systemPrompt }] },
                    generationConfig: { responseMimeType: "application/json" }
                })
            });

            const result = await response.json();
            const subtasks = JSON.parse(result.candidates[0].content.parts[0].text);
            
            subtaskContainer.innerHTML = '';
            subtasks.forEach(st => {
                subtaskContainer.innerHTML += `
                    <div class="flex items-center gap-2 text-sm subtask-item">
                        <span>•</span> <span>${st}</span>
                    </div>
                `;
            });
        } catch (error) {
            subtaskContainer.innerHTML = '<span class="text-xs text-gray-400 italic">Keine Häppchen verfügbar.</span>';
        }
    }

    function copyToNotion(id) {
        const taskEl = document.getElementById(`task-${id}`);
        const title = taskEl.querySelector('.task-title').innerText;
        const subtasks = Array.from(taskEl.querySelectorAll('.subtask-item')).map(s => s.innerText.replace('• ', ''));
        
        // Formatierung für Notion: Titel gefolgt von Checkboxen ([] )
        let text = `[] ${title}\n`;
        subtasks.forEach(st => {
            text += `  [] ${st}\n`;
        });

        const el = document.createElement('textarea');
        el.value = text;
        document.body.appendChild(el);
        el.select();
        document.execCommand('copy');
        document.body.removeChild(el);

        const btn = taskEl.querySelector('button');
        btn.innerText = '✅ Copied!';
        setTimeout(() => btn.innerText = '📋 Copy for Notion', 2000);
    }
</script>
